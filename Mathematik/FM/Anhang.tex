\begin{equation}
\unit{N}=\unit{kg\,m/s^2}.
\end{equation}

\begin{equation}
\unit{W}=\unit{kg\,m^2/s^3}=\unit{VA}.
\end{equation}

\begin{equation}
\unit{J}=\unit{kg\,m^2/s^2}=\unit{Nm}=\unit{Ws}=\unit{VAs}.
\end{equation}

\begin{equation}
\unit{Pa}=\unit{N/m^2} = 10^{-5}\,\unit{bar}.
\end{equation}
Hertz (Frequenz):
\begin{equation}
\unit{Hz} = \unit{1/s}.
\end{equation}
Coulomb (Ladung):
\begin{equation}
\unit{C} = \unit{As}.
\end{equation}
Volt (Spannung):
\begin{equation}
\unit{V} = \unit{kg\,m^2/(A\,s^3)}
\end{equation}

\begin{equation}
\unit{T} = \unit{N/(A\,m)} = \unit{Vs/m^2}.
