Potenzfunktion

// Naiver Algorithmus

long pow(long a, unsigned int n){
  long y=1;
  for(unsigned int i=0; i<n; i++){
    y*=a;
  }
  return y;
}

// Expandiert man den folgenden kompakten Algorithmus,
// so merkt man, dass der Exponent in eine binäre Darstellung
// zerlegt wird.

long pow(long a, unsigned int n){
  long y=1;
  while(n>0){
    if(n%2==0){
      n/=2; a*=a;
    }else{
      n--; y*=a;
    }
  }
  return y;
}

// Da Zahlen schon in binärer Darstellung gespeichert sind,
// ist die Division n/2 eigentlich nicht notwendig.
// Man kann n/2 gegen n>>1 und n%2 gegen n&1 ersetzen,
// die Bedingung n>0 lässt sich gegen n ersetzen,
// was aber alles vom optimierenden Compiler automatisch gemacht
// werden sollte.

long pow(long a, unsigned int n){
  long y=1;
  while(n){
    if(n&1){
      n--; y*=a;
    }else{
      n>>=1; a*=a;
    }
  }
  return y;
}

// Modulare Potenzfunktion
// y = a^n mod m

typedef unsigned long ulong;

ulong powmod(ulong a, ulong n, ulong m){
  ulong y=1;
  a=a%m;
  while(n>0){
    if(n%2==0){
      n/=2; a=(a*a)%m;
    }else{
      n--; y=(y*a)%m;
    }
  }
  return y;
}

Binomialkoeffizient

long choose(long n, long k){
  long y,i;
  if(k>n) return 0;
  if(k>n-k) k=n-k;
  y=1; k--;
  for(i=0; i<=k; i++){
    y = y*(n-i)/(i+1);
  }
  return y;
}

// Pascalsches Dreieck

void Pascal(){
  for(long i=0; i<=12; i++){
    for(long j=0; j<=i; j++){
      printf("%4i",choose(i,j));
    }
    printf("\n");
  }
}

Größter gemeinsamer Teiler

// Euklidischer Algorithmus

long gcd(long a, long b){
  if(b==0) return a;
  else if(a==0) return b;
  else if(a>b) return gcd(a-b,b);
  else return gcd(a,b-a);
}

long gcd(long a, long b){
  if(a==0) return b;
  while(b!=0){
    if(a>b) a=a-b;
    else b=b-a;
  }
  return a;
}

long gcd(long a, long b){
  if(b==0) return a;
  else return gcd(b,a%b);
}

long gcd(long a, long b){
  long h;
  while(b!=0){
    h = a%b; a=b; b=h;
  }
  return a;
}

Primzahlen

// Naiver Primzahltest

int isprime(long n){
  long i;
  for(i=2; i<n; i++){
    if(n%i==0) return 0;
  }
  return 1;
}


// Verbesserter Primzahltest

int isprime(long n){
  long i;
  for(i=2; i*i<=n; i++){
    if(n%i==0) return 0;
  }
  return 1;
}


// Fermatscher Primzahltest

int fermat(long n){
  long i;
  if(n<6){
    return isprime(n);
  }
  for(i=2; i<6; i++){
    if(powmod(i,n-1,n)!=1) return 0;
  }
  return 1;
}

// Mischen

void shuffle(long* a, long n){
  long i,j,h;
  for(i=n-1; i>0; i--){
    j=rand()%(i+1);
    h=a[j];
    a[j]=a[i];
    a[i]=h;
  }
}

Sortierung

Bubblesort

typedef long Element;

void swap(Element* a, Element* b){
  Element h;
  h=*b; *b=*a; *a=h;
}

void bubblesort(Element* a, size_t n){
  size_t i,j;
  for(j=0; j<n; j++){
    for(i=0; i+1<n; i++){
      if(a[i+1]<a[i]) swap(a+i,a+i+1);
    }
  }
}

Quicksort

size_t partition(Element* a, size_t n, size_t k){
  Element pivot = a[k];
  swap(a+k,a+n-1);
  size_t i=0;
  for(size_t j=0; j+1<n; j++){
    if(a[j]<=pivot){
      swap(a+i,a+j);
      i++;
    }
  }
  swap(a+i,a+n-1);
  return i;
}

void quicksort(Element* a, size_t n){
  if(n<2) return;
  size_t i = partition(a,n,0);
  quicksort(a,i);
  quicksort(a+i+1,n-i-1);
}


// Quicksort, polymorph

void memory_swap(char* a, char* b, int size){
  char h;
  for(int i=0; i<size; i++){
    h=b[i]; b[i]=a[i]; a[i]=h;
  }
}

void quicksort(void* p, int n, int size, int compare(void*,void*)){
  int i,j=0;
  char* a = (char*)p;
  if(n<2) return;
  memory_swap(a+n/2*size,a+(n-1)*size,size);
  for(i=0; i<n-1; i++){
    if(compare(a+i*size,a+(n-1)*size)<=0){
      memory_swap(a+i*size,a+j*size,size);
      j++;
    }
  }
  memory_swap(a+j*size,a+(n-1)*size,size);
  quicksort(a,j,size,compare);
  quicksort(a+(j+1)*size,n-j-1,size,compare);
}

int compare_int(void* a, void* b){
  return *(int*)a-*(int*)b;
}

void sort(int* a, int n){
  qsort(a,n,sizeof(int),compare_int);
}

Zeichenketten

// Umwandlung einer Zeichenkette
// in eine Zahl.

int stoi(const char* s){
  int i,y,sgn;
  i=0; y=0; sgn=1;
  if(s[i]=='-'){
    i++; sgn=-1;
  }
  while(isdigit(s[i])){
    y=10*y+s[i]-48;
    i++;
  }
  return sgn*y;
}

Suchalgorithmen

Teilzeichenketten

// Suche von einer Teilzeichenkette
// in einer Zeichenkette.
// Naiver Algorithmus.

int find(const char* a, int n, const char* sub, int subn){
  int i,j,m,found;
  m=n-subn+1;
  for(j=0; j<m; j++){
    found=1;
    for(i=0; i<subn; i++){
      if(a[j+i]!=sub[i]){
        found=0;
        break;
      }
    }
    if(found) return j;
  }
  return -1;
}

Binäre Suche

int bfind(const int* p, int n, int x){
  int a,b,m;
  a=0; b=n-1;
  while(a<=b){
    m = (a+b)/2;
    if(p[m]==x) return m;
    if(p[m]<x) a=m+1;
    else b=m-1;
  }
  return -1;
}

// Binäre Suche, polymorph

int bfind(const void* data, int n, int size, void* x,
  int compare(const void*,const void*)
){
  const char* p = (const char*)data;
  int a,b,m;
  a=0; b=n-1;
  while(a<=b){
    m = (a+b)/2;
    if(compare(p+m*size,x)==0) return m;
    if(compare(p+m*size,x)<0) a=m+1;
    else b=m-1;
  }
  return -1;
}

int compare_int(const void* a, const void* b){
  return *(int*)a-*(int*)b;
}

int find(const int* a, int n, int x){
  return bfind(a,n,sizeof(int),&x,compare_int);
}
