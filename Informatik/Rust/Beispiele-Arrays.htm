Array umdrehen

fn reverse(a: &mut [i32]){
    let n = a.len();
    for i in 0..n/2 {
        a.swap(i,n-1-i);
    }
}

// Typ-generisch:
fn reverse<T>(a: &mut [T]){
    let n = a.len();
    for i in 0..n/2 {
        a.swap(i,n-1-i);
    }
}

fn main(){
    let mut a: Vec<i32> = vec![1,2,3,4];
    reverse(&mut a);
    println!("{:?}",a);
}

Teilarrays für gerade und ungerade Indizes

fn take_even(a: &[i32]) -> Vec<i32> {
    let mut v: Vec<i32> = Vec::new();
    let n = a.len();
    let m = if n%2==0 {n/2} else {n/2+1};
    for i in 0..m {
        v.push(a[2*i]);
    }
    return v;
}

fn take_odd(a: &[i32]) -> Vec<i32> {
    let mut v: Vec<i32> = Vec::new();
    for i in 0..a.len()/2 {
        v.push(a[2*i+1]);
    }
    return v;
}

fn main(){
    let a: Vec<i32> = vec![1,2,3,4];
    let v = take_even(&a);
    let w = take_odd(&a);
    println!("{:?}, {:?}",v,w);
}
