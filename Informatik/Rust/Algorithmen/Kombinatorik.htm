Iterativ

fn power_set<T: Clone>(a: impl IntoIterator<Item=T>) -> Vec<Vec<T>> {
    let mut ps = vec![vec![]];
    for x in a {
        for mut s in ps.clone() {
            s.push(x.clone());
            ps.push(s);
        }
    }
    return ps;
}
