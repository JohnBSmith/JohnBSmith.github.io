Größter gemeinsamer Teiler

// Klassischer euklidischer Algorithmus, rekursiv
fn gcd(a: u32, b: u32) -> u32 {
    return if b==0 {a}
      else if a==0 {b}
      else if b<a {gcd(a-b,b)}
      else {gcd(a,b-a)};
}

// Klassischer euklidischer Algorithmus, iterativ
fn gcd(mut a: u32, mut b: u32) -> u32 {
    if a==0 {
        return b;
    }else{
        while b != 0 {
            if b<a {a = a-b;} else {b = b-a;}
        }
        return a;
    }
}

// Moderner euklidischer Algorithmus, rekursiv
fn gcd(a: u32, b: u32) -> u32 {
    return if b==0 {a} else {gcd(b,a%b)};
}

// Moderner euklidischer Algorithmus, iterativ
// ("Turbo-Euklid")
fn gcd(mut a: u32, mut b: u32) -> u32 {
    while b != 0 {
        let h = a%b;
        a = b; b = h;
    }
    return a;
}

Kleinstes gemeinsames Vielfaches

fn lcm(a: u32, b: u32) -> u32 {
    return a*b/gcd(a,b);
}

fn lcm_slice(a: &[u32]) -> u32 {
    return a.iter().fold(1,|a,&b| {a*b/gcd(a,b)});
}

fn is_prime(n: u64) -> bool {
    let mut i: u64 = 2;
    while i*i<=n {
        if n%i==0 {return false;}
        i+=1;
    }
    return true;
}

Schneller Primzahltest

fn is_prime_trial_div(n: u32) -> bool {
    if n<4 {return n>1;}
    if n%2==0 || n%3==0 {return false;}
    let mut i = 5;
    let mut w = 2;
    while i*i<=n {
        if n%i==0 {return false;}
        i += w;
        w = 6-w;
    }
    return true;
}

fn pow_mod(a: u32, mut n: u32, m: u32) -> u64 {
    let mut p: u64 = 1;
    let mut a: u64 = u64::from(a%m);
    while n>1 {
        if n&1 == 1 {p = (p*a)%u64::from(m);}
        n >>= 1; a = (a*a)%u64::from(m);
    }
    if n==1 {p = (p*a)%u64::from(m);}
    return p;
}

fn witness(a: u32, d: u32, n: u32, mut r: u32) -> bool {
    let mut p = pow_mod(a,d,n);
    if p==1 {
        return true;
    } else {
        while r != 0 {
            if p == u64::from(n-1) {return true;}
            p = (p*p)%u64::from(n);
            r -= 1;
        }
        return false;
    }
}

pub fn is_prime(n: u32) -> bool {
    if n<13000000 {
        return is_prime_trial_div(n);
    } else if n&1==0 {
        return false;
    }

    // let (2^r)*d = n-1;
    let mut r = 0;
    let mut d = n-1;
    while d&1==0 {r += 1; d >>= 1;}

    // Miller-Rabin-Test
    return [2,7,61].iter().all(|&a| witness(a,d,n,r));
}

Primfaktorzerlegung

fn factor(mut n: u64) -> Vec<(u64,u8)> {
    let mut v: Vec<(u64,u8)> = Vec::new();
    let mut i = 2;
    while i<=n {
        let mut p = 0;
        while n%i==0 {n = n/i; p += 1;}
        if p != 0 {v.push((i,p));}
        i += 1;
    }
    return v;
}

// Teilerliste
fn divisors(n: u64) -> Vec<u64> {
    return (1..n+1).filter(|i| n%i==0).collect();
}

// Teileranzahlfunktion
fn sigma0(n: u64) -> usize {
    return (1..n+1).filter(|i| n%i==0).count();
}

// Teilerfunktion
fn sigma(x: u32, n: u64) -> u64 {
    return (1..n+1).filter(|i| n%i==0).map(|i| i.pow(x)).sum();
}
