Bubblesort

fn bubble_sort<T: Ord>(a: &mut [T]) {
    let n = a.len();
    for _ in 0..n {
        for i in 0..n-1 {
            if a[i+1]<a[i] {a.swap(i,i+1);}
        }
    }
}

fn main() {
    let mut a = [3,1,4,2];
    bubble_sort(&mut a);
    println!("{:?}",a);
}
