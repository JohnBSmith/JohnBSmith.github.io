Theorem. iota_property
x = ⋂{y |P y} → (∃x. set x ∧ ∀y. x = y ↔ P y) → P x
Proof
01. 1 ⊢ x = ⋂{y |P y}, hypo.
02. 2 ⊢ ∃x. set x ∧ ∀y. x = y ↔ P y, hypo.
03. 2 ⊢ ∃x. set x ∧ {x} = {y | P y}, sg_eq_from_ex_uniq 2.
04. 4 ⊢ set a ∧ {a} = {y | P y}, hypo.
05. 4 ⊢ set a, conj_eliml 4.
06. 4 ⊢ {a} = {y | P y}, conj_elimr 4.
07. 4 ⊢ ⋂{a} = ⋂{y | P y}, f_equal 6, f t = ⋂t.
08. 1, 4 ⊢ x = ⋂{a}, eq_trans_rr 1 7.
09. 4 ⊢ ⋂{a} = a, Intersection_sg 5.
10. 1, 4 ⊢ x = a, eq_trans 8 9.
11. 1, 4 ⊢ x ∈ {a}, sg_intro 5 10.
12. 1, 4 ⊢ x ∈ {y | P y}, eq_subst 6 11, P t ↔ x ∈ t.
13. 1, 4 ⊢ P x, comp_elim 12.
14. 1, 2 ⊢ P x, ex_elim 3 13.
iota_property. ⊢ x = ⋂{y |P y} → (∃x. set x ∧ ∀y. x = y ↔ P y) → P x,
  subj_intro_ii 14.
